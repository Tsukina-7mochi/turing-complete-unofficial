SLTi t0 amount t1
BEQ zero t1 LOAD_END
LX zero INOUT t1
SX t0 RAM t1
ADDi t0 1 t0
J _ _ LOAD_LOOP
label LOAD_END

# bubble sort start
# loop index: s0
ADD zero zero s0
label BSORT_OUTER
# loop index: s1
ADDi s0 1 s1

label BSORT_INNER
# load and compare
LX s0 RAM t0
LX s1 RAM t1
SLTU t0 t1 t2
BNE t2 zero SWAP_END

#swap
SX s0 RAM t1
SX s1 RAM t0
label SWAP_END
#swap end

label BSORT_INNER_END

BEQ t0 zero BSORT_OUTER_END

J _ _ BSORT_OUTER

label BSORT_OUTER_END
#bubble sort end
